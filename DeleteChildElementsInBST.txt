In an Array of BST, If an elemnt is given to be deleteds, all the child elements in the list are to be deleted and only the remaining elements are left out 

//Note: We use 2*i +1 , 2*i + 2 to determine the position of the elements child elements
//Implemented a recursion to delete the elements  

import scala.collection.mutable.ArrayBuffer

class bufferElements( ) {
  
   def findElemList(size: Int, elem: Int, arr: Array[Int] ): ArrayBuffer[Int] = {
     
    
      val pos = BSearch(size, elem, arr, 0)
     
      println(" pos "+ pos)
     
      val arrBuff = new ArrayBuffer[Int] 
      
     arr.foreach{ i => arrBuff += i }
     
     return finalList(arrBuff: ArrayBuffer[Int],size: Int, pos: Int, arr: Array[Int] )
   }
  
  def finalList(arrBuff: ArrayBuffer[Int], size: Int, pos: Int, arr: Array[Int] ): ArrayBuffer[Int] = {
         if( pos == -1) { return arrBuff}
        var AB: ArrayBuffer[Int] = arrBuff
        
        if( 2*pos +2 > size && 2*pos +1 == size  ) { 
                   
            AB -= arr(2*pos +1)
            AB -= arr(pos)
            return AB
        }
        
         if( 2*pos +1 > size  ) { 
                   
            AB -= arr(pos)
            return AB
        }
        
        finalList(AB, size: Int,2*pos +1, arr )
        finalList(AB, size: Int,2*pos +2, arr )
      
      AB -= arr(pos)        
        return AB
        
      }
  
  
  def BSearch(end: Int, elem: Int, arr: Array[Int], start: Int ): Int = {
    
       if( start > end ) { return -1}
       
       val mid: Int = (start + end )/2
       
        if(arr(mid) == elem ){ return mid }
    
        if( arr(mid) > elem ){
          BSearch(mid-1, elem, arr, start)
          
        }
        else { BSearch(end, elem, arr, mid+1) }   
      
    
  }
  
}
