Find duplicate:

Approach 1: using map 

object findDuplicate {
   
    var map = Map[Int, Int ]()
  def dupCheck( arr: Array[Int] ): Int =  {
    arr.foreach(x => {
      if( map.contains(x) ) { return x}
      map = map + (x -> x)
    })
    return 0
  }
  
  def main(args: Array[String]): Unit = {
    
    val arr = Array( 1, 2, 3, 4, 2  )
    println("Duplicate in the array is "+ dupCheck(arr) ) 
  }
  
}
